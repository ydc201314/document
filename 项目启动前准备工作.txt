1.开启zookeeper
	[root@itcast ~]# cd /usr/local/zookeeper-3.4.6/bin/
	[root@itcast bin]# ./zkServer.sh start
2.开启单机版redis
	[root@itcast bin]# cd /usr/local/redis_alone/bin
	[root@itcast bin]# ./redis-server redis.conf
3.开启集群版solr
	[root@itcast bin]# cd /usr/local/solr/solr-cloud/zookeeper01/bin/
	[root@itcast bin]# ./zkServer.sh start

	[root@itcast bin]# cd /usr/local/solr/solr-cloud/zookeeper02/bin/
	[root@itcast bin]# ./zkServer.sh start

	[root@itcast bin]# cd /usr/local/solr/solr-cloud/zookeeper03/bin/
	[root@itcast bin]# ./zkServer.sh start

	[root@itcast bin]#  cd /usr/local/solr/solr-cloud/tomcat-solr01/bin/
	[root@itcast bin]# ./startup.sh

	[root@itcast bin]#  cd /usr/local/solr/solr-cloud/tomcat-solr02/bin/
	[root@itcast bin]# ./startup.sh

	[root@itcast bin]#  cd /usr/local/solr/solr-cloud/tomcat-solr03/bin/
	[root@itcast bin]# ./startup.sh

	[root@itcast bin]#  cd /usr/local/solr/solr-cloud/tomcat-solr04/bin/
	[root@itcast bin]# ./startup.sh
4.开启activeMQ
	[root@itcast bin]# cd /usr/local/apache-activemq-5.12.0/bin/linux-x86-32/
	[root@itcast linux-x86-32]# ./activemq start
5.woindowns下开启Nginx服务器
